## Adding Your First Component

Use the `bct add` command to add components to your project:

```bash
npx bct add button
```

This command will:
1. Download the button component from the BCT registry
2. Place it in `src/components/` (or `components/` if no src directory)
3. Install any required dependencies
4. Make the component ready to use

## Using Components

After adding a component, import and use it in your application:

```tsx
import { Button } from '@/components/button'

export default function MyPage() {
  return (
    <div className="p-8">
      <h1 className="text-3xl font-bold mb-4">Welcome</h1>
      <Button variant="primary" onClick={() => alert('Clicked!')}>
        Click Me
      </Button>
    </div>
  )
}
```

## Building a Simple Form

Let's build a complete form using multiple BCT UI components:

### Step 1: Add Required Components

```bash
npx bct add button
npx bct add input
npx bct add checkbox
```

### Step 2: Create the Form Component

```tsx
import { useState } from 'react'
import { Button } from '@/components/button'
import { TextInput } from '@/components/input'
import { Checkbox } from '@/components/checkbox'

export default function SignupForm() {
  const [email, setEmail] = useState('')
  const [password, setPassword] = useState('')
  const [agreed, setAgreed] = useState(false)

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
    console.log({ email, password, agreed })
  }

  return (
    <form onSubmit={handleSubmit} className="max-w-md mx-auto p-6 bg-surface-1 rounded-lg">
      <h2 className="text-2xl font-bold mb-6">Sign Up</h2>
      
      <div className="mb-4">
        <label className="block text-sm font-medium mb-2">Email</label>
        <TextInput
          type="email"
          value={email}
          onChange={(e) => setEmail(e.target.value)}
          placeholder="you@example.com"
          required
        />
      </div>
      
      <div className="mb-4">
        <label className="block text-sm font-medium mb-2">Password</label>
        <TextInput
          type="password"
          value={password}
          onChange={(e) => setPassword(e.target.value)}
          placeholder="‚Ä¢‚Ä¢‚Ä¢‚Ä¢‚Ä¢‚Ä¢‚Ä¢‚Ä¢"
          required
        />
      </div>
      
      <div className="mb-6">
        <Checkbox
          checked={agreed}
          onChange={(e) => setAgreed(e.target.checked)}
          label="I agree to the terms and conditions"
        />
      </div>
      
      <Button type="submit" variant="primary" className="w-full">
        Create Account
      </Button>
    </form>
  )
}
```

## Common Patterns

### Loading States

```tsx
import { Button } from '@/components/button'
import { Spinner } from '@/components/spinner'

function LoadingButton({ isLoading, children, ...props }) {
  return (
    <Button {...props} disabled={isLoading}>
      {isLoading ? <Spinner size="sm" /> : children}
    </Button>
  )
}
```

### Form Validation

```tsx
import { TextInput } from '@/components/input'
import { Alert } from '@/components/alert'

function ValidatedInput({ error, ...props }) {
  return (
    <div>
      <TextInput {...props} className={error ? 'border-error' : ''} />
      {error && (
        <Alert variant="error" className="mt-2">
          {error}
        </Alert>
      )}
    </div>
  )
}
```

### Confirmation Dialog

```tsx
import { AlertDialog } from '@/components/alert-dialog'
import { Button } from '@/components/button'

function DeleteConfirmation({ onConfirm, onCancel }) {
  return (
    <AlertDialog
      title="Delete Item"
      description="Are you sure? This action cannot be undone."
      actions={
        <>
          <Button variant="tertiary" onClick={onCancel}>
            Cancel
          </Button>
          <Button variant="primary" onClick={onConfirm}>
            Delete
          </Button>
        </>
      }
    />
  )
}
```

## Styling Components

BCT UI components use Tailwind CSS classes and design tokens:

```tsx
// Using design tokens
<Button className="bg-primary text-primary-on">
  Primary Button
</Button>

// Combining with Tailwind utilities
<Card className="p-6 shadow-md hover:shadow-lg transition-shadow">
  Card content
</Card>

// Responsive styling
<div className="p-4 md:p-6 lg:p-8">
  Responsive padding
</div>
```
